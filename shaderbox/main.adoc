{

        data[i] = 0;
    });
}

{

[source, cpp]
-----
template<size_t i, typename Other, size_t Other_N>
void construct_at_index(vector<Other, Other_N>&& arg)
{
    constexpr auto count = std::min(i + Other_N, num_components);

    static_for<i, count>()([&](size_t j) {
        data[j] = arg.data[j - i];
    });
}
